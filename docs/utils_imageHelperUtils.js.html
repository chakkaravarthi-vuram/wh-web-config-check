/**
 * @function base64StringtoFile
 * @description To convert base 64 string into file
 * @param   {string} base64String base 64 string format
 * @param   {string} filename file name
 * @return  file
 */
 export const base64StringtoFile = (base64String, filename) => {
    const arr = base64String.split(',');
    const mime = arr[0].match(/:(.*?);/)[1];
    const bstr = atob(arr[1]);
    const n = bstr.length;
    const u8arr = new Uint8Array(n);
    while (n - 1) {
      u8arr[n] = bstr.charCodeAt(n);
    }
    return new File([u8arr], filename, { type: mime });
  };
/**
 * @memberof imagehelper

  /**
   * @function downloadBase64File
   * @description To download base 64 file
   * @param   {string} base64Data base 64 string format
   * @param   {string} filename file name
   */
  export const downloadBase64File = (base64Data, filename) => {
    const element = document.createElement('a');
    element.setAttribute('href', base64Data);
    element.setAttribute('download', filename);
    element.style.display = 'none';
    document.body.appendChild(element);
    element.click();
    document.body.removeChild(element);
  };

  /**
   * @function image64toCanvasRef
   * @description To convert base 64 string to  canvas ref
   * @param {string} canvasRef ref for canvas element
   * @param {string} image64 image in base 64 format
   * @param {string} pixelCrop number of pixels to crop
   */
  export const image64toCanvasRef = (canvasRef, image64, pixelCrop) => {
    const canvas = canvasRef;
    canvas.width = pixelCrop.width;
    canvas.height = pixelCrop.height;
    const ctx = canvas.getContext('2d');
    const image = new Image();
    image.src = image64;
    const scaleX = image.naturalWidth / image.width;
    const scaleY = image.naturalHeight / image.height;
    image.onload = () => {
      ctx.drawImage(
        image,
        pixelCrop.x * scaleX,
        pixelCrop.y * scaleY,
        pixelCrop.width * scaleX,
        pixelCrop.height * scaleY,
        0,
        0,
        pixelCrop.width,
        pixelCrop.height,
      );
    };
  };

  /**
   * @function extractImageFileExtensionFromBase64
   * @description To extract extensiuon of  image from base64
   * @param {string} base64Data base 64 string format
   * @return string
   */

  export const extractImageFileExtensionFromBase64 = (base64Data) => {
    const base64 = base64Data;
    base64.substring('data:image/'.length, base64.indexOf(';base64'));
  };
